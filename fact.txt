
#include <stdio.h>
void main()
{
    int i=1,fact=1,num;
    printf("enter the number");
    scanf("%d",&num);
    while(i<=num)
    {
        fact=fact*i;
        i++;
    }
    printf("factorialof %d is a %d",num,fact);
}
